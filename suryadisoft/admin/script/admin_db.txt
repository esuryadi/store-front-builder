CREATE TABLE USER (user_id VARCHAR (99)  NOT NULL , password VARCHAR (99)  NOT NULL , first_name VARCHAR (99) , last_name VARCHAR (99) , email VARCHAR (99) , role SET('Administrator','User','Sales') NOT NULL , status SET ('Active','Inactive','Suspended') NOT NULL , sales_id VARCHAR(99) , PRIMARY KEY(user_id))

CREATE TABLE CLIENTS (user_id VARCHAR (99)  NOT NULL , company_name VARCHAR (99)  NOT NULL , company_url VARCHAR (99)  NOT NULL , company_address1 TEXT, company_address2 TEXT, company_city VARCHAR (99) , company_state VARCHAR (99) , company_zip VARCHAR (12) , company_country VARCHAR (99) , company_phone VARCHAR (99) , company_fax VARCHAR (99) , company_email VARCHAR (99) , PRIMARY KEY(user_id))
	
CREATE TABLE CLIENT_COMPONENTS (id INT UNSIGNED  NOT NULL  AUTO_INCREMENT , user_id VARCHAR (99)  NOT NULL , component VARCHAR (50)  NOT NULL , PRIMARY KEY(id))

CREATE TABLE CLIENT_DATABASE (user_id VARCHAR (99)  NOT NULL , database_name VARCHAR (99)  NOT NULL , PRIMARY KEY(user_id))

CREATE TABLE THEMES_PROPERTY (theme_name VARCHAR (99)  NOT NULL , theme_color_scheme VARCHAR(99) NOT NULL , theme_property_name VARCHAR (200)  NOT NULL , theme_property_value VARCHAR (200)  NOT NULL )

CREATE TABLE THEMES_COLOR_SCHEME (theme_name VARCHAR (99)  NOT NULL , theme_color_scheme VARCHAR (99)  NOT NULL )

CREATE TABLE THEMES (theme_name VARCHAR (99)  NOT NULL , theme_directory VARCHAR (200)  NOT NULL , theme_filename VARCHAR (200)  NOT NULL , PRIMARY KEY(theme_name))

CREATE TABLE CLIENT_PAYMENT_SERVICE (user_id VARCHAR (99)  NOT NULL , payment_service VARCHAR (99)  NOT NULL , PRIMARY KEY(user_id))

CREATE TABLE PRICING (pricing_id INT UNSIGNED  NOT NULL  AUTO_INCREMENT , pricing_plan VARCHAR (200)  NOT NULL , pricing_category VARCHAR (200)  NOT NULL , pricing_price DECIMAL (9,2)  NOT NULL , pricing_term SET ('One Time','Monthly')  NOT NULL , PRIMARY KEY(pricing_id))

CREATE TABLE BUILT_IN_WEB_CONTENT (id INT UNSIGNED NOT NULL AUTO_INCREMENT, component_name VARCHAR(99) not null , title TEXT, filename TEXT, type ENUM ('Frame','No Frame') , position SET ('Top','Bottom','Left','Center','Right') , sequence INT UNSIGNED , display_name VARCHAR(255), description TEXT, PRIMARY KEY(id))

CREATE TABLE COMPONENT_DESIGN (component_name VARCHAR (99)  NOT NULL , design_style VARCHAR (99)  NOT NULL , filename VARCHAR (99)  NOT NULL , preview_images VARCHAR (99)  NOT NULL )

CREATE TABLE BILLING (user_id VARCHAR (99)  NOT NULL , billing_first_name VARCHAR (99)  NOT NULL , billing_last_name VARCHAR (99)  NOT NULL , billing_address_1 VARCHAR (200) , billing_address_2 VARCHAR (200) , billing_city VARCHAR (99) , billing_state VARCHAR (99) , billing_zip VARCHAR (20), billing_country VARCHAR(99) , billing_phone VARCHAR (20) , payment_type SET ('Visa','Master Card','Discover','American Express')  NOT NULL , account_number VARCHAR (99)  NOT NULL , cc_exp_date VARCHAR (6)  NOT NULL , monthly_fee DECIMAL (9,2)  NOT NULL , order_date DATETIME NOT NULL , PRIMARY KEY(user_id), INDEX(account_number))

CREATE TABLE TRIAL_ORDER (id INT UNSIGNED  NOT NULL  AUTO_INCREMENT , user_id VARCHAR (99)  NOT NULL , first_name VARCHAR (99)  NOT NULL , last_name VARCHAR (99)  NOT NULL , address_1 TEXT NOT NULL , address_2 TEXT NOT NULL , city VARCHAR (99)  NOT NULL , state VARCHAR (99)  NOT NULL , zip VARCHAR (10)  NOT NULL , country VARCHAR (99)  NOT NULL , phone VARCHAR (20)  NOT NULL , email VARCHAR (99)  NOT NULL , order_date DATE NOT NULL , status SET ('Active','Expired')  NOT NULL , build_status SET('pending','completed') , PRIMARY KEY(id), INDEX(user_id))

CREATE TABLE STATE (short_name VARCHAR (2)  NOT NULL , long_name VARCHAR (99)  NOT NULL , country VARCHAR (99)  NOT NULL , PRIMARY KEY(short_name), INDEX(short_name))

CREATE TABLE SALES_TAX (state VARCHAR (2)  NOT NULL , sales_tax_rate DECIMAL (1,5)  NOT NULL , PRIMARY KEY(state), INDEX(state))

CREATE TABLE COMPONENT_PROPERTIES (component_name VARCHAR (99)  NOT NULL , property_name VARCHAR (99)  NOT NULL , property_type VARCHAR (99)  NOT NULL , property_default_value TEXT , property_option_values VARCHAR (200), property_description TEXT)

CREATE TABLE REFERRAL (referral_id VARCHAR (99)  NOT NULL , first_name VARCHAR (99) NOT NULL , middle_initial VARCHAR(5) , last_name VARCHAR (99) NOT NULL , address_1 VARCHAR (200) , address_2 VARCHAR (200) , city VARCHAR (99) , state VARCHAR (99) , zip VARCHAR (99) , country VARCHAR (99) , phone VARCHAR (99) , paid_amount DECIMAL(9,2) , PRIMARY KEY(referral_id) , INDEX(last_name))

CREATE TABLE BILLING_RECORD (id INT UNSIGNED  NOT NULL  AUTO_INCREMENT , user_id VARCHAR (99)  NOT NULL , date DATETIME, description VARCHAR (99) , amount DECIMAL (9,2)  NOT NULL , status SET ('paid','unpaid'), PRIMARY KEY(id), INDEX(id))

CREATE TABLE SALES_ASSOCIATE (sales_id VARCHAR (99) NOT NULL , sales_first_name VARCHAR (200) NOT NULL , sales_middle_initial VARCHAR (1) , sales_last_name VARCHAR (200) NOT NULL , sales_address_1 TEXT NOT NULL , sales_address_2 TEXT, sales_city VARCHAR (99) NOT NULL , sales_state VARCHAR (99) NOT NULL , sales_zip VARCHAR (10) NOT NULL , sales_country VARCHAR (99) , sales_home_phone VARCHAR (15) NOT NULL , sales_business_phone VARCHAR (15) , sales_cellular_phone VARCHAR (15) , sales_fax VARCHAR (15) , sales_email VARCHAR(200) , sales_ssn VARCHAR (11) NOT NULL , sales_dob DATE NOT NULL , sales_married_status ENUM ('Single','Married') , sales_commission DECIMAL (1,4) NOT NULL , PRIMARY KEY(sales_id), INDEX(sales_id))
